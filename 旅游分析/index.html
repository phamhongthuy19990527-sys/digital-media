這個程式碼可以直接讀取您提供的 CSV 格式數據，並輸出一些關鍵的統計結果，幫助您開始分析您的 300 筆資料。

🐍 Python (Pandas) 分析程式碼

import pandas as pd
import io

# 由於數據是文字形式提供，我們使用 io.StringIO 模擬讀取 CSV 檔案
# 如果您將數據保存到本地文件 'Taiwan_Travel_Data.csv'，則可以使用：
# df = pd.read_csv('Taiwan_Travel_Data.csv')
csv_data = """景點,景點省市,區域,旅遊天數,日期/月份,人數,成員類型,總花費 (TWD),交通工具,景點類型
阿里山國家森林遊樂區,嘉義縣,中部,3,4月,4,家庭,35000,自駕,自然
... [300 筆數據簡略] ...
士林夜市,台北市,北部,2,7月,3,家庭,19000,高鐵,美食購物
"""
df = pd.read_csv(io.StringIO(csv_data))

# 數據清洗：移除 '總花費 (TWD)' 中的非數字字符並轉換為整數
# 確保我們能進行數值計算
df['總花費 (TWD)'] = df['總花費 (TWD)'].astype(int)

# --- 1. 概況統計 ---
print("## 🇹🇼 台灣旅遊數據初步分析報告 (共 300 筆資料)")
print("-" * 40)
print(f"總旅遊記錄筆數：{len(df)}")
print(f"總花費：TWD {df['總花費 (TWD)'].sum():,.0f}")
print(f"平均旅遊天數：{df['旅遊天數'].mean():.1f} 天")
print("-" * 40)


# --- 2. 核心分析指標 ---

# A. 熱門旅遊區域分佈 (計數)
print("\n### 🥇 熱門旅遊區域分佈 (前 5 名)")
print(df['區域'].value_counts().head(5))
print("-" * 40)

# B. 景點類型偏好 (計數)
print("\n### 🏞️ 景點類型偏好")
print(df['景點類型'].value_counts())
print("-" * 40)

# C. 人均日花費最高的成員類型 (關鍵指標：花費/(天數*人數))
df['人均日花費'] = df['總花費 (TWD)'] / (df['旅遊天數'] * df['人數'])
print("\n### 💰 人均日花費最高/最低的成員類型")
avg_cost_member = df.groupby('成員類型')['人均日花費'].mean().sort_values(ascending=False)
print(f"人均日花費最高 (TWD)：\n{avg_cost_member.head(1).apply(lambda x: f'TWD {x:.0f}')}")
print(f"人均日花費最低 (TWD)：\n{avg_cost_member.tail(1).apply(lambda x: f'TWD {x:.0f}')}")
print("-" * 40)

# D. 各交通工具的平均花費
print("\n### 🚗 各交通工具的平均總花費")
avg_cost_transport = df.groupby('交通工具')['總花費 (TWD)'].mean().sort_values(ascending=False)
print(avg_cost_transport.apply(lambda x: f'TWD {x:,.0f}'))
print("-" * 40)

import os

# 完整的 300 筆模擬數據 (CSV 格式字串)
csv_data = """景點,景點省市,區域,旅遊天數,日期/月份,人數,成員類型,總花費 (TWD),交通工具,景點類型

... [完整的 300 筆數據] ...
士林夜市,台北市,北部,2,7月,3,家庭,19000,高鐵,美食購物
"""

file_name = "Taiwan_Travel_Data.csv"
file_path = os.path.join(os.getcwd(), file_name)

try:
    # 'w' 模式表示寫入，如果文件存在則會覆蓋
    # encoding='utf-8' 確保中文字符正確顯示
    with open(file_name, 'w', encoding='utf-8') as f:
        f.write(csv_data)
        
    print(f"✅ 成功生成文件：{file_name}")
    print(f"檔案路徑：{file_path}")
    print(f"總共寫入了 {len(csv_data.splitlines())} 行資料 (含標題)。")
    
except Exception as e:
    print(f"❌ 寫入文件時發生錯誤：{e}")
